```python
import pandas as pd
from tqdm import tqdm
import json

def batch_extract_skills(csv_file, output_file):
    """批量提取技能"""
    df = pd.read_csv(csv_file)
    
    all_skills = []
    
    for idx, row in tqdm(df.iterrows(), total=len(df)):
        job_desc = row['岗位描述']
        
        # 提取技能
        skills = extract_skills_with_llm(job_desc)
        
        if skills:
            all_skills.append({
                '岗位名称': row['岗位名称'],
                '技能数据': json.dumps(skills, ensure_ascii=False)
            })
        
        # 每100条保存一次
        if (idx + 1) % 100 == 0:
            temp_df = pd.DataFrame(all_skills)
            temp_df.to_csv(output_file, index=False)
    
    # 最终保存
    final_df = pd.DataFrame(all_skills)
    final_df.to_csv(output_file, index=False)
    
    return final_df

# 运行
```

```python
from collections import Counter

def generate_skill_dictionary(extracted_file):
    """从提取结果生成词典"""
    df = pd.read_csv(extracted_file)
    
    all_skills = Counter()
    skill_categories = {}
    
    for skills_json in df['技能数据']:
        skills_data = json.loads(skills_json)
        
        for category, skills in skills_data.items():
            if category == '熟练度':
                continue
            
            for skill in skills:
                all_skills[skill] += 1
                skill_categories[skill] = category
    
    # 生成词典文件
    with open('dicts/userdict_zh_recruitment_extended.txt', 'w', encoding='utf-8') as f:
        for skill, count in all_skills.most_common():
            # 格式：词汇 词频 词性
            freq = min(count * 100, 20000)  # 根据出现次数设置词频
            f.write(f"{skill} {freq} nz\n")
    
    print(f"生成扩展词典：{len(all_skills)} 个技能")
    
    # 生成分类词典
    with open('dicts/skill_taxonomy.json', 'w', encoding='utf-8') as f:
        taxonomy = {}
        for skill, category in skill_categories.items():
            if category not in taxonomy:
                taxonomy[category] = []
            taxonomy[category].append(skill)
        json.dump(taxonomy, f, ensure_ascii=False, indent=2)
    
    return all_skills, skill_categories

# 运行
skills, categories = generate_skill_dictionary('output/llm_extracted_skills.csv')
```

```python
# 加载中文NER模型
model_name = "ckiplab/bert-base-chinese-ner"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForTokenClassification.from_pretrained(model_name)

# 创建NER pipeline
ner_pipeline = pipeline(
    "ner",
    model=model,
    tokenizer=tokenizer,
    device=0  # 使用GPU 0
)

def extract_skills_bert(text):
    """使用BERT提取技能"""
    results = ner_pipeline(text)
    
    skills = []
    current_skill = ""
    
    for item in results:
        if item['entity'].startswith('B-'):  # 开始
            if current_skill:
                skills.append(current_skill)
            current_skill = item['word']
        elif item['entity'].startswith('I-'):  # 继续
            current_skill += item['word']
        else:  # 结束
            if current_skill:
                skills.append(current_skill)
                current_skill = ""
    
    if current_skill:
        skills.append(current_skill)
    
    return skills
```

```python
# 查找相似词
def find_similar_skills(skill, topn=20):
    """查找相似技能"""
    try:
        similar = model.wv.most_similar(skill, topn=topn)
        return [word for word, score in similar if score > 0.5]
    except:
        return []

# 扩展词典
base_skills = ['Python', 'Java', 'JavaScript', '数据分析', '机器学习']
extended_skills = set(base_skills)

for skill in base_skills:
    similar_skills = find_similar_skills(skill)
    extended_skills.update(similar_skills)
    print(f"{skill} 相似技能: {similar_skills}")
```

```python
# 使用cupy加速向量计算
import cupy as cp

# 将词向量转移到GPU
vectors_gpu = cp.array(model.wv.vectors)

def fast_similar_search(word_vector, topn=20):
    """GPU加速的相似度搜索"""
    word_vector_gpu = cp.array(word_vector)
    
    # 计算余弦相似度
    similarities = cp.dot(vectors_gpu, word_vector_gpu)
    similarities /= cp.linalg.norm(vectors_gpu, axis=1)
    similarities /= cp.linalg.norm(word_vector_gpu)
    
    # 获取top-k
    top_indices = cp.argsort(similarities)[-topn:]
    
    return top_indices.get()  # 转回CPU
```

```python
class SkillExtractorPipeline:
    """技能提取流水线"""
    
    def __init__(self):
        # 加载Word2Vec
        self.w2v_model = Word2Vec.load('models/word2vec_skills.model')
        
        # 加载BERT-NER
        self.ner_pipeline = pipeline("ner", model="ckiplab/bert-base-chinese-ner", device=0)
        
        # 加载大模型（按需）
        self.llm_tokenizer = None
        self.llm_model = None
    
    def extract_stage1_w2v(self, base_skills):
        """阶段1：Word2Vec快速扩展"""
        extended = set(base_skills)
        
        for skill in base_skills:
            similar = self.find_similar_skills(skill, topn=10)
            extended.update(similar)
        
        return list(extended)
    
    def extract_stage2_bert(self, texts):
        """阶段2：BERT批量提取"""
        all_skills = set()
        
        for text in tqdm(texts):
            skills = self.extract_with_bert(text)
            all_skills.update(skills)
        
        return list(all_skills)
    
    def extract_stage3_llm(self, important_texts):
        """阶段3：大模型精细处理"""
        if not self.llm_model:
            self.load_llm()
        
        detailed_skills = {}
        
        for text in tqdm(important_texts):
            skills = self.extract_with_llm(text)
            detailed_skills[text] = skills
        
        return detailed_skills
    
    def run_pipeline(self, df):
        """运行完整流水线"""
        print("阶段1：Word2Vec扩展基础词典...")
        base_skills = ['Python', 'Java', 'JavaScript', '数据分析', '机器学习']
        stage1_skills = self.extract_stage1_w2v(base_skills)
        print(f"  扩展到 {len(stage1_skills)} 个技能")
        
        print("\n阶段2：BERT批量提取...")
        texts = df['岗位描述'].tolist()
        stage2_skills = self.extract_stage2_bert(texts)
        print(f"  提取到 {len(stage2_skills)} 个技能")
        
        print("\n阶段3：大模型精细处理高薪岗位...")
        high_salary_df = df[df['平均薪资'] > 20000]
        important_texts = high_salary_df['岗位描述'].tolist()[:1000]  # 只处理1000个
        stage3_skills = self.extract_stage3_llm(important_texts)
        print(f"  精细提取 {len(stage3_skills)} 个岗位")
        
        # 合并结果
        all_skills = set(stage1_skills) | set(stage2_skills)
        for skills_dict in stage3_skills.values():
            for category, skills in skills_dict.items():
                if isinstance(skills, list):
                    all_skills.update(skills)
        
        print(f"\n最终技能数：{len(all_skills)}")
        
        return list(all_skills)

# 使用
pipeline = SkillExtractorPipeline()
df = pd.read_csv('output/nlp_processed/前程无忧_NLP处理_样本_1%.csv')
final_skills = pipeline.run_pipeline(df)
```

```python
def batch_extract(texts, batch_size=8):
    """批量处理"""
    results = []
    for i in range(0, len(texts), batch_size):
        batch = texts[i:i+batch_size]
        batch_results = model.generate(batch)
        results.extend(batch_results)
    return results
```
